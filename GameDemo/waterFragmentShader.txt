#version 330 core

in vec2 textureCoords;			//UV position

out vec4 out_Colour;

uniform sampler2D reflectionTexture;
uniform sampler2D refractionTexture;

void main(void){

	vec4 reflectionColor = texture(reflectionTexture, textureCoords);
	vec4 refractionColor = texture(refractionTexture, textureCoords);

	out_Colour = mix(reflectionColor, refractionColor, 0.5);
}
